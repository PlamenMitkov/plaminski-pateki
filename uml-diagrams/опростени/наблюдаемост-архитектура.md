```mermaid
graph TB
    subgraph Приложение["EcoProject Приложение"]

        Serilog["Serilog"]
        Конзола["Конзолен изход\n(Docker logs)"]

    end

    subgraph Мониторинг["Мониторинг"]
        Здраве["Health Check\nGET /health"]
        Метрики["Персонализирани метрики\n(rate limit, AI fallback)"]
    end

    А --> Serilog
    Serilog --> Конзола
    Serilog --> Файл
    А --> Здраве
    А --> Метрики
```
