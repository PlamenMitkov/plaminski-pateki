classDiagram

        +Id : string
        +Имейл : string

    class ЛюбимМаршрут {
        +ПотребителId : string
        +МаршрутId : int

    }

        +Id : int
        +СесияId : string

    class АиСъобщение {
        +Id : int
        +СесияId : int
        +Роля : string
        +Съдържание : string

    }

    class ОбщностенПост {
        +Id : int
        +ПотребителId : string

    Потребител "1" --> "*" ЛюбимМаршрут
    Маршрут "1" --> "*" ЛюбимМаршрут

    АиСесия "1" --> "*" АиСъобщение
    Потребител "1" --> "*" ОбщностенПост
